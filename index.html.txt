    <script>
        let usdjpyPrice = 154.50;
        let accountType = 'standard';

        async function updateExchangeRate() {
            try {
                // 使用极其稳定的开源汇率接口
                const res = await fetch('https://open.er-api.com/v6/latest/USD');
                const data = await res.json();
                if(data && data.rates && data.rates.JPY) {
                    usdjpyPrice = parseFloat(data.rates.JPY);
                    document.getElementById('usdjpyRate').innerText = `USDJPY ${usdjpyPrice.toFixed(2)}`;
                    const dot = document.getElementById('statusDot');
                    dot.classList.replace('bg-yellow-500', 'bg-green-500');
                    dot.classList.remove('animate-pulse');
                }
            } catch (e) {
                document.getElementById('usdjpyRate').innerText = "STABLE MODE";
            }
            calculate();
        }

        function setAccount(type) {
            accountType = type;
            document.getElementById('btn-standard').className = type === 'standard' ? "flex-1 py-2 text-xs font-bold rounded-xl transition-all tab-active" : "flex-1 py-2 text-xs font-bold rounded-xl transition-all text-slate-500";
            document.getElementById('btn-cent').className = type === 'cent' ? "flex-1 py-2 text-xs font-bold rounded-xl transition-all tab-active" : "flex-1 py-2 text-xs font-bold rounded-xl transition-all text-slate-500";
            document.getElementById('accountTypeLabel').innerText = type === 'cent' ? "Cent A/C (x100)" : "Standard A/C";
            calculate();
        }

        const inputs = ['riskAmount', 'symbol', 'entryPrice', 'stopPrice', 'splitCount'];
        inputs.forEach(id => {
            document.getElementById(id).addEventListener('input', () => {
                if(id === 'splitCount') document.getElementById('splitValue').innerText = document.getElementById(id).value;
                calculate();
            });
        });

        function calculate() {
            const risk = parseFloat(document.getElementById('riskAmount').value) || 0;
            const symbol = document.getElementById('symbol').value;
            const entry = parseFloat(document.getElementById('entryPrice').value) || 0;
            const stop = parseFloat(document.getElementById('stopPrice').value) || 0;
            const splits = parseInt(document.getElementById('splitCount').value);

            if (!risk || !entry || !stop || entry === stop) return;

            const priceDiff = Math.abs(entry - stop);
            let totalLots = 0;
            let note = "";

            switch(symbol) {
                case "XAUUSD":
                    totalLots = risk / (priceDiff * 100);
                    note = `Dist: ${priceDiff.toFixed(2)} USD`;
                    break;
                case "XAGUSD":
                    totalLots = risk / (priceDiff * 5000);
                    note = `Dist: ${priceDiff.toFixed(3)} USD`;
                    break;
                case "BTCUSD":
                case "ETHUSD":
                    totalLots = risk / priceDiff;
                    note = `Gap: ${priceDiff.toFixed(2)} Pts`;
                    break;
                case "EURJPY":
                    const pipValueInUSD = 1000 / usdjpyPrice; 
                    const pips = priceDiff * 100;
                    totalLots = risk / (pips * pipValueInUSD);
                    note = `${pips.toFixed(1)} Pips Dist`;
                    break;
            }

            if (accountType === 'cent') {
                totalLots = totalLots * 100;
            }

            const each = totalLots / splits;

            document.getElementById('resultLots').innerText = totalLots.toFixed(2);
            document.getElementById('eachLot').innerText = each.toFixed(2);
            document.getElementById('detailText').innerText = note;
        }

        updateExchangeRate();
        setInterval(updateExchangeRate, 3600000); // 稳定版每小时核对一次汇率即可
    </script>
